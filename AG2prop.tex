        \subsection{Divisors on curves}

        Let $X$ be regular integral projective scheme of dim 1 over a field $k$\\

        \begin{theorem}
          If $D$ is the effective divisor associated to a fin closed subscheme:
        \begin{enumerate}
        \item $T \subset X$ then $D = \sum_{P \in T} length(\ca{O}_{T,P} \cdot P)$
          \item and $deg_k(D) = \sum_{P \in T} lenth(\ca{O}_{T,P}) \cdot [k(P)/k] = dim_k\Gamma(T, \ca{O}_T) \overset{§4.8}{=:} deg(T)$\\
        \end{enumerate}        
        \end{theorem}


        \begin{theorem}
          If $k'$ is a fin extension of $k$ $\Rightarrow$ $deg_k(D) = deg_{k'}(D) \cdot [k'/k]$\\
        \end{theorem}
        
        
        \begin{theorem}
          For any field extension $k'/k$ let $X':= X_{k'} = X \times_{Spec(k)} Spec(k') \xrightarrow{\pi} X$\\
          Assume $X'$ integral then $deg_{k'}(\pi^*D) = deg_k(D)$\\
        \end{theorem}
